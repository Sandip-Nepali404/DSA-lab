## Algorithm :
- push()

1. If 










       Tos = max-1 Then display "Stack is Full and Stop"



2. Else 


       2.1 Read data 


       2.2 Tos <- Tos+1

       2.3 stack[TOS] <- data


- pop()

1.  If




        Tos = -1 Then display "Stack is Empty and Stop"






2.  Else 










        data <- stack[TOS]
        Tos<-Tos-1
        Return data
